CREATE TABLE IF NOT EXISTS projects (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  name TEXT NOT NULL,
  description TEXT,
  keywords TEXT NOT NULL,       -- JSON array e.g. ["Metformin","adverse"]
  sources TEXT NOT NULL,        -- JSON array e.g. ["twitter","reddit"]
  latency TEXT NOT NULL,        -- JSON object e.g. {"twitter":"realtime","reddit":"daily"}
  status TEXT DEFAULT 'active', -- active | paused
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  signal_count INTEGER DEFAULT 0
);
